CREATE TABLE users (
    user_id INT AUTO_INCREMENT,
    username VARCHAR(50) UNIQUE,
    name VARCHAR(50),
    email VARCHAR(100) UNIQUE,
    passwrd VARCHAR(100),
    phone_number VARCHAR(20),
    gender VARCHAR(10),
    country_id INT,
    birthday TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    profile_pic VARCHAR(50) DEFAULT 'default.png',
    website_url VARCHAR(80),

    usr_status BOOLEAN,
    acc_status BOOLEAN DEFAULT TRUE,
    blood_group_id INT,
    is_admin BOOLEAN DEFAULT FALSE,

    donated_at TIMESTAMP,
    created_at TIMESTAMP,
    updated_at TIMESTAMP,

    PRIMARY KEY (user_id),
    FOREIGN KEY (blood_group_id) REFERENCES blood_groups(group_id),
    FOREIGN KEY (country_id) REFERENCES countries(country_id),
    FOREIGN KEY (usr_status) REFERENCES statuses(status_id),
    FOREIGN KEY (acc_status) REFERENCES statuses(status_id)
);


CREATE TABLE posts (
    post_id INT AUTO_INCREMENT,
    user_id INT,
    post_status INT,
    blood_group_id INT,
    blood_process_id INT,
    post_type VARCHAR(10),
    post_blood_unit INT,
    post_city VARCHAR(50),
    post_location VARCHAR(125),
    post_created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    post_updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    PRIMARY KEY (post_id),
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (post_status) REFERENCES statuses(status_id),
    FOREIGN KEY (blood_group_id) REFERENCES blood_groups(group_id),
    FOREIGN KEY (blood_process_id) REFERENCES blood_processes(process_id)
);


CREATE TABLE statuses (
    status_id  INT AUTO_INCREMENT,
    status_type VARCHAR(20),

    PRIMARY KEY (status_id)
);


CREATE TABLE countries (
    country_id  INT AUTO_INCREMENT,
    country_name VARCHAR(60),

    PRIMARY KEY (country_id)
);


CREATE TABLE reports (
    report_id INT AUTO_INCREMENT,
    report_reason VARCHAR(40),
    post_id INT,

    PRIMARY KEY (report_id),
    FOREIGN KEY (post_id) REFERENCES posts(post_id)
);


CREATE TABLE blood_groups (
    group_id INT AUTO_INCREMENT,
    group_name VARCHAR(3),

    PRIMARY KEY (group_id)
);


CREATE TABLE diseases (
    disease_id INT AUTO_INCREMENT,
    disease_name VARCHAR(50),

    PRIMARY KEY (disease_id)
);


CREATE TABLE blood_processes (
    process_id INT AUTO_INCREMENT,
    process_name VARCHAR(30),

    PRIMARY KEY (process_id)
);


CREATE TABLE user_diseases (
    user_id INT,
    disease_id INT,

    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (disease_id) REFERENCES diseases(disease_id)
);

CREATE TABLE post_diseases (
    post_id INT,
    disease_id INT,

    FOREIGN KEY (post_id) REFERENCES posts(post_id),
    FOREIGN KEY (disease_id) REFERENCES diseases(disease_id)
);
