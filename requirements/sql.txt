CREATE TABLE users (
    user_id INT(16) AUTO_INCREMENT,
    name VARCHAR(50),
    username VARCHAR(50) UNIQUE,
    email VARCHAR(100) UNIQUE,
    passwrd VARCHAR(100),
    phone_number VARCHAR(20),
    website_url VARCHAR(80),
    birthday TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    gender VARCHAR(10),
    country VARCHAR(60),
    profile_pic VARCHAR(250),
    blood_group_id INT(16),
    last_donated TIMESTAMP,
    status BOOLEAN DEFAULT TRUE,
    is_admin BOOLEAN DEFAULT FALSE,

    PRIMARY KEY (user_id),
    FOREIGN KEY (blood_group_id) REFERENCES blood_groups(group_id)
);


CREATE TABLE posts (
    post_id INT(16) AUTO_INCREMENT,
    user_id INT(16),
    blood_group_id INT(16),
    blood_process_id INT(16),
    post_type VARCHAR(10),
    post_blood_unit INT(16),
    post_city VARCHAR(50),
    post_location VARCHAR(125),
    post_status VARCHAR(15) DEFAULT 'pending',
    post_created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    post_updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    PRIMARY KEY (post_id),
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (blood_group_id) REFERENCES blood_groups(group_id),
    FOREIGN KEY (blood_process_id) REFERENCES blood_processes(process_id)
);


CREATE TABLE reports(
    report_id INT(16) AUTO_INCREMENT,
    report_reason VARCHAR(40),
    post_id INT(16),

    PRIMARY KEY (report_id),
    FOREIGN KEY (post_id) REFERENCES posts(post_id)
);


CREATE TABLE blood_groups (
    group_id INT(16) AUTO_INCREMENT,
    group_name VARCHAR(3),

    PRIMARY KEY (group_id)
);

CREATE TABLE disease_types (
    d_type_id INT(16) AUTO_INCREMENT,
    d_type_name VARCHAR(50),

    PRIMARY KEY (d_type_id)
);

CREATE TABLE blood_processes (
    process_id INT(16) AUTO_INCREMENT,
    process_name VARCHAR(30),

    PRIMARY KEY (process_id)
);


CREATE TABLE user_posts (
    user_id INT(16),
    post_id INT(16),

    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (post_id) REFERENCES posts(post_id)
);


CREATE TABLE user_diseases (
    user_id INT(16),
    disease_id INT(16),
    
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (disease_id) REFERENCES disease_types(d_type_id)
);
